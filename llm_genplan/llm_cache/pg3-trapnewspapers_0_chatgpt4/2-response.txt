def get_plan(objects, init, goal):
    home_base = None
    safe_locations = set()
    unpacked_papers = set()
    loc_wants_paper = set()
    plan = []

    # Extract information from init and objects
    for predicate in init:
        if predicate[0] == 'ishomebase':
            home_base = predicate[1]
        elif predicate[0] == 'safe':
            safe_locations.add(predicate[1])
        elif predicate[0] == 'unpacked':
            unpacked_papers.add(predicate[1])
        elif predicate[0] == 'wantspaper':
            loc_wants_paper.add(predicate[1])

    # Iterate through locations that want a paper
    for loc in loc_wants_paper:
        # Pick up a newspaper
        paper = unpacked_papers.pop()
        plan.append(f'(pick-up {paper} {home_base})')

        # Move to the destination location (assuming direct movement to the location)
        plan.append(f'(move {home_base} {loc})')

        # Deliver the newspaper
        plan.append(f'(deliver {paper} {loc})')

    return plan
