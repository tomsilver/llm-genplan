def get_plan(objects, init, goal):
    # Extract weight relationships
    weight_relationships = [atom for atom in init if atom[0] == "heavier"]

    # Create a dictionary to store the weight relationships
    weight_dict = {}
    for relation in weight_relationships:
        item1, item2 = relation[1], relation[2]
        if item1 not in weight_dict:
            weight_dict[item1] = set()
        weight_dict[item1].add(item2)

    # Sort items in descending order of their weight
    sorted_items = sorted(weight_dict.keys(), key=lambda x: len(weight_dict[x]), reverse=True)

    # Initialize the plan
    plan = []

    # Pack the heaviest item first
    plan.append(f"(pack-first {sorted_items[0]})")

    # Stack items in descending order of their weight
    for i in range(1, len(sorted_items)):
        plan.append(f"(stack {sorted_items[i - 1]} {sorted_items[i]})")

    return plan
